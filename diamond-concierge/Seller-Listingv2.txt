// ── STEPPER ──
function renderStepper() {
  document.getElementById('stepBar').innerHTML = STEPS.map((lbl,i)=>{
    const n=i+1, st=n<step?'done':n===step?'active':'pending', num=st==='done'?'✓':n;
    const sep = i<STEPS.length-1 ? '<span class="stp-connector"></span>' : '';
    return `<div class="stp" onclick="goStep(${n})"><div class="stp-num ${st}">${num}</div><span class="stp-lbl ${st}">${lbl}</span></div>${sep}`;
  }).join('');
  document.getElementById('progressLine').style.width = ((step-1)/(STEPS.length-1)*100)+'%';
}

// ── NAVIGATION ──
function goStep(n) {
  if(n<1||n>5) return;
  document.getElementById('panel-'+step).classList.remove('active');
  step=n;
  document.getElementById('panel-'+step).classList.add('active');
  renderStepper(); updateScore(); updateFooter();
  // show/hide right panel cards
  const scoreCard = document.getElementById('scoreCard');
  const pricingRightCard = document.getElementById('pricingRightCard');
  if(n===4) { scoreCard.style.display='none'; pricingRightCard.style.display='block'; renderPricing(); updatePricingRightCard(); }
  else { scoreCard.style.display='block'; pricingRightCard.style.display='none'; }
  if(n===2) { renderThumbs(); renderTagChips(); }
  if(n===5) updateReview();
  renderSections();
  scrollTo({top:0,behavior:'smooth'});
}
function nextStep(){ step===5 ? handlePublish() : goStep(step+1); }
function prevStep(){ goStep(step-1); }

function updateFooter() {
  document.getElementById('btnPrev').style.visibility = step===1 ? 'hidden' : 'visible';
  const btn = document.getElementById('btnNext');
  if(step===5){ btn.innerHTML='Publish Listing ✦'; btn.classList.add('publish'); }
  else { btn.innerHTML='Continue <span style="opacity:.4;font-size:11px;margin-left:2px">→</span>'; btn.classList.remove('publish'); }
}
function renderSections() {
  setTimeout(()=>{
    document.querySelectorAll('#panel-'+step+' .section:not(.vis)').forEach((s,i)=>
      setTimeout(()=>s.classList.add('vis'),100*(i+1)));
  },40);
}

// ── TAGS ──
function initTags() {
  renderTagChips();
  const inp = document.getElementById('tagIn');
  if(!inp) return;
  inp.addEventListener('keydown',e=>{
    if((e.key==='Enter'||e.key===',')&&inp.value.trim()){ e.preventDefault(); addTag(inp.value.replace(/,/g,'').trim()); inp.value=''; }
    if(e.key==='Backspace'&&!inp.value&&tags.length){ tags.pop(); renderTagChips(); }
  });
  inp.addEventListener('blur',()=>{ if(inp.value.trim()){ addTag(inp.value.replace(/,/g,'').trim()); inp.value=''; } });
}
function addTag(v){ if(!v||tags.includes(v)||tags.length>=12) return; tags.push(v); renderTagChips(); }
function removeTag(i){ tags.splice(i,1); renderTagChips(); }
function renderTagChips() {
  const c = document.getElementById('tagsChips'); if(!c) return;
  c.innerHTML = tags.map((t,i)=>
    `<span class="tag-chip" onclick="removeTag(${i})">${t}<svg width="9" height="9" viewBox="0 0 9 9" style="flex-shrink:0"><line x1="2" y1="2" x2="7" y2="7" stroke="currentColor" stroke-width="1.5"/><line x1="7" y1="2" x2="2" y2="7" stroke="currentColor" stroke-width="1.5"/></svg></span>`
  ).join('');
}

// ── PHOTO UPLOAD ──
function initUpload() {
  const outer = document.getElementById('uploadOuter');
  if(!outer) return;
  outer.addEventListener('dragover',e=>{e.preventDefault();outer.classList.add('drag-over')});
  outer.addEventListener('dragleave',()=>outer.classList.remove('drag-over'));
  outer.addEventListener('drop',e=>{
    e.preventDefault(); outer.classList.remove('drag-over');
    Array.from(e.dataTransfer.files).filter(f=>f.type.startsWith('image/')).forEach(readPhoto);
  });
}
function onFileSelect(inp){ Array.from(inp.files).forEach(readPhoto); inp.value=''; }
function readPhoto(f) {
  if(photos.length>=10) return;
  const r = new FileReader();
  r.onload = e=>{ photos.push({url:e.target.result,name:f.name}); renderThumbs(); };
  r.readAsDataURL(f);
}
function removePhoto(i){ photos.splice(i,1); renderThumbs(); }
function renderThumbs() {
  const grid = document.getElementById('thumbGrid');
  const zone = document.getElementById('uploadZone');
  const section = document.getElementById('thumbSection');
  const outer = document.getElementById('uploadOuter');
  if(!grid) return;
  if(!photos.length) {
    zone.style.display='flex'; section.style.display='none';
    outer.classList.remove('has-photos'); return;
  }
  zone.style.display='none'; section.style.display='block';
  outer.classList.add('has-photos');
  document.getElementById('thumbCount').textContent = `${photos.length} photo${photos.length!==1?'s':''}`;
  grid.innerHTML = photos.map((p,i)=>`
    <div class="thumb">
      <img src="${p.url}" style="width:100%;height:100%;object-fit:cover;position:absolute;inset:0" alt="">
      <div class="thumb-ov">
        <button class="thumb-rm" onclick="removePhoto(${i})">
          <svg width="10" height="10" viewBox="0 0 10 10"><line x1="2" y1="2" x2="8" y2="8" stroke="white" stroke-width="1.5"/><line x1="8" y1="2" x2="2" y2="8" stroke="white" stroke-width="1.5"/></svg>
        </button>
      </div>
      ${i===0?'<div class="thumb-cover">Cover</div>':''}
    </div>`).join('');
}

function onPrice(el) {
  const v = parseFloat(el?.value||0);
  const fb = document.getElementById('priceFb'); if(!fb) return;
  if(isNaN(v)||v<=0){ fb.innerHTML='<span class="fb-bad">Please enter a valid price</span>'; return; }
  fb.innerHTML = v>5200 ? '<span class="fb-warn">⚠ Above recommended range — may take longer to sell</span>'
    : v<3900 ? '<span class="fb-good">✓ Priced for a fast sale</span>'
    : '<span class="fb-good">✓ Competitive price — strong visibility</span>';
  const payout=Math.round(v*.92), fee=Math.round(v*.08);
  document.getElementById('payoutVal').textContent='$'+payout.toLocaleString();
  document.getElementById('feeVal').textContent='$'+fee.toLocaleString();
  // update right panel
  const prcPayout=document.getElementById('prcPayout');
  const prcFee=document.getElementById('prcFee');
  const prcVel=document.getElementById('prcVelocity');
  if(prcPayout) prcPayout.textContent='$'+payout.toLocaleString();
  if(prcFee) prcFee.textContent=`After 8% GemGem fee · − $${fee.toLocaleString()}`;
  if(prcVel) {
    const days = v>5200 ? '45–70 days' : v<3900 ? '3–7 days' : '12–18 days';
    const tip = v>5200 ? 'Consider lowering to sell faster.' : v<3900 ? 'Great for a quick, clean sale.' : 'Competitive pricing — solid balance.';
    prcVel.innerHTML = `Stones at this price typically sell in <strong style="color:rgba(255,255,255,.7)">${days}</strong>. ${tip}`;
  }
}

let chatTyping = false;
function initChat() {
  addBotMsg("Hi! I'm your GemGem diamond advisor. I can help you value your stone, optimise your price, and answer any listing questions.\n\nWhat would you like to know?");
}
function addBotMsg(text) {
  const msgs=document.getElementById('chatMsgs');
  const div=document.createElement('div'); div.className='msg msg-bot';
  div.innerHTML=`<div class="msg-bubble">${text.replace(/\*\*(.*?)\*\*/g,'<strong>$1</strong>').replace(/\n/g,'<br>')}</div><div class="msg-time">just now</div>`;
  msgs.appendChild(div); msgs.scrollTop=msgs.scrollHeight;
}
function addUserMsg(text) {
  const msgs=document.getElementById('chatMsgs');
  const div=document.createElement('div'); div.className='msg msg-user';
  div.innerHTML=`<div class="msg-bubble">${text}</div><div class="msg-time">just now</div>`;
  msgs.appendChild(div); msgs.scrollTop=msgs.scrollHeight;
}
function showTyping() {
  const msgs=document.getElementById('chatMsgs');
  const div=document.createElement('div'); div.className='msg msg-bot'; div.id='typingMsg';
  div.innerHTML='<div class="msg-bubble" style="padding:0"><div class="typing-bubble"><div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div></div></div>';
  msgs.appendChild(div); msgs.scrollTop=msgs.scrollHeight;
}
function hideTyping(){ const t=document.getElementById('typingMsg'); if(t) t.remove(); }
function sendChat() {
  const inp=document.getElementById('chatIn');
  const q=inp.value.trim(); if(!q||chatTyping) return;
  inp.value=''; chatTyping=true;
  addUserMsg(q); showTyping();
  const lq=q.toLowerCase();
  let respFn=CHAT_RESPONSES['default'];
  for(const key of Object.keys(CHAT_RESPONSES)){
    if(key!=='default'&&lq.includes(key)){ respFn=CHAT_RESPONSES[key]; break; }
  }
  setTimeout(()=>{ hideTyping(); const resp=typeof respFn==='function'?respFn(q):respFn; addBotMsg(resp); chatTyping=false; },900+Math.random()*600);
}
function askQ(btn) {
  document.getElementById('chatIn').value=btn.textContent;
  sendChat();
  document.getElementById('chatQs').style.display='none';
}
